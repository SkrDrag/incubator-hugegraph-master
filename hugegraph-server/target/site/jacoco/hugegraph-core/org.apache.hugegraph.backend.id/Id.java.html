package org.apache.hugegraph.backend.id;

import org.apache.hugegraph.util.E;

public interface Id extends Comparable<Id> {

    int UUID_LENGTH = 16;

    Object asObject();

    String asString();

    long asLong();

    byte[] asBytes();

    int length();

    IdType type();

    default boolean number() {
        return this.type() == IdType.LONG;
    }

    default boolean uuid() {
        return this.type() == IdType.UUID;
    }

    default boolean string() {
        return this.type() == IdType.STRING;
    }

    default boolean edge() {
        return this.type() == IdType.EDGE;
    }

    enum IdType {

        UNKNOWN,
        LONG,
        UUID,
        STRING,
        EDGE;

        public char prefix() {
            if (this == UNKNOWN) {
                return 'N';
            }
            return this.name().charAt(0);
        }

        public static IdType valueOfPrefix(String id) {
            E.checkArgument(id != null && id.length() > 0,
                            "Invalid id '%s'", id);
            switch (id.charAt(0)) {
                case 'L':
                    return IdType.LONG;
                case 'U':
                    return IdType.UUID;
                case 'S':
                    return IdType.STRING;
                case 'E':
                    return IdType.EDGE;
                default:
                    return IdType.UNKNOWN;
            }
        }
    }
}
